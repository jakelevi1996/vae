\section{VI intro: EM algorithm (setup)}

\begin{itemize}
    \item VAE builds on VI
    \item EM = good conceptual intro to VI
    \item Problem: suppose we have a probabilistic model with IID observed data $x_i$, latent variables $z_i$, and parameters $\theta$ for $i\in\{1,\dots,N\}$:
    \begin{align}
        p\bigl(x_i, z_i \,\vert\, \theta\bigr) &= p\bigl( z_i \,\vert\, \theta \bigr)\;p\bigl( x_i \,\vert\, z_i, \theta \bigr) && \text{(Joint/generative model)}
    \end{align}
    \item EG textbook example = GMM:
    \begin{align}
        p\bigl(x_i, z_i \,\vert\, \theta\bigr) &= \pi_{(z_i)}\;\mathcal{N}\left( x_i \;\middle\vert\; \mu_{(z_i)}, \sigma_{(z_i)}^2 \right) && \text{(GMM joint distribution)}
    \end{align}
    \item Want to find parameters $\theta^*$ that maximise likelihood of observed data (minimise NLL):
    \begin{align}
        \theta^* &= \underset{\theta}{\text{argmin}}\Bigl[ \mathcal{L}(x, \theta) \Bigr] && \text{(Optimal parameters)} \\
        \mathcal{L}(x, \theta) &= -\log\bigl( p\bigl(x \,\vert\, \theta\bigr) \bigr) && \text{(Loss function: NLL)} \\
        &= -\sum_{i=1}^n \left[ \log\Bigl( p\bigl(x_i \,\vert\, \theta\bigr) \Bigr) \right] && \text{(Sum over data points)} \\
        &= -\sum_{i=1}^n \left[ \log\left( \sum_{z_i} \Bigl[ p\bigl(x_i, z_i \,\vert\, \theta\bigr) \Bigr] \right) \right] && \text{(Marginalise over latents)} \label{eq:EM loss function}
    \end{align}
    \item Assume easy to:
    \begin{enumerate}
        \item Find latent posterior $p\bigl( z_i \,\vert\, x_i, \theta \bigr)$
        \item Find $\theta^*$ which maximises $p\bigl(x, z \,\vert\, \theta\bigr)$ when $z$ are known (don't need to marginalise)
    \end{enumerate}
    \item BUT, in general, no closed-form solution to Equation \ref{eq:EM loss function} (because of sum inside log)
\end{itemize}
